on: [pull_request]

jobs:
  autoprefix-to-pr:
    runs-on: ubuntu-latest
    name: autoprefixToPR
    if: ${{ github.event.pull_request.base.ref == 'main' && github.event.pull_request.merged == true }}
    permissions:
      pull-requests: write
    steps:
      - name: Checkout
        uses: actions/checkout@v3
        with:
          fetch-depth: 0
      - name: Set autoprefix
        uses: ./autoprefix-to-pr
        with:
          repo-token: ${{ secrets.GITHUB_TOKEN }}
